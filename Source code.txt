#include "main.h"
#include <stdio.h>
#include <string.h>

ADC_HandleTypeDef hadc1;
UART_HandleTypeDef huart2;

uint32_t soil_moisture;
uint32_t light_value;
uint32_t gas_value;
uint8_t relay_status = 0;

char tx_buffer[200];

// Function Prototypes
void SystemClock_Config(void);
static void MX_GPIO_Init(void);
static void MX_ADC1_Init(void);
static void MX_USART2_UART_Init(void);

uint32_t Read_ADC(uint32_t channel);

int main(void)
{
    HAL_Init();
    SystemClock_Config();

    MX_GPIO_Init();
    MX_ADC1_Init();
    MX_USART2_UART_Init();

    while (1)
    {
        // Read sensors
        soil_moisture = Read_ADC(ADC_CHANNEL_0);
        light_value   = Read_ADC(ADC_CHANNEL_1);
        gas_value     = Read_ADC(ADC_CHANNEL_2);

        // Decision logic for irrigation
        if (soil_moisture > 800)   // Dry soil
        {
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_SET); // Pump ON
            relay_status = 1;
        }
        else
        {
            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_5, GPIO_PIN_RESET); // Pump OFF
            relay_status = 0;
        }

        // Send data to ESP32 via UART
        sprintf(tx_buffer,
                "Soil:%lu Light:%lu Gas:%lu Pump:%d\r\n",
                soil_moisture, light_value, gas_value, relay_status);

        HAL_UART_Transmit(&huart2, (uint8_t*)tx_buffer, strlen(tx_buffer), HAL_MAX_DELAY);

        HAL_Delay(2000);
    }
}

uint32_t Read_ADC(uint32_t channel)
{
    ADC_ChannelConfTypeDef sConfig = {0};

    sConfig.Channel = channel;
    sConfig.Rank = ADC_REGULAR_RANK_1;
    sConfig.SamplingTime = ADC_SAMPLETIME_71CYCLES_5;

    HAL_ADC_ConfigChannel(&hadc1, &sConfig);

    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, HAL_MAX_DELAY);

    return HAL_ADC_GetValue(&hadc1);
}
